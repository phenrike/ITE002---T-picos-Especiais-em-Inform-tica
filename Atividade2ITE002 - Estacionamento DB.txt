CREATE DATABASE ESTACIONAMENTO;

USE ESTACIONAMENTO;

CREATE TABLE CARRO (
  CARRO_ID BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  MODELO VARCHAR(100) NOT NULL,
  PLACA VARCHAR(100) NOT NULL,
  CONSTRAINT CARRO_PK PRIMARY KEY (CARRO_ID)
);

CREATE TABLE CLIENTE (
  CLIENTE_ID BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  CARRO_ID BIGINT UNSIGNED NOT NULL,
  NOME VARCHAR(100) NOT NULL,
  CONSTRAINT CLIENTE_PK PRIMARY KEY (CLIENTE_ID),
  FOREIGN KEY CARRO_FK (CARRO_ID) REFERENCES CARRO (CARRO_ID) ON DELETE RESTRICT ON UPDATE CASCADE
);

CREATE TABLE ESTACIONAMENTO (
  CLIENTE_ID BIGINT UNSIGNED NOT NULL,
  PRIMARY KEY (CLIENTE_ID),
  FOREIGN KEY CLIENTE_FK (CLIENTE_ID) REFERENCES CLIENTE (CLIENTE_ID) ON DELETE RESTRICT ON UPDATE CASCADE
);
